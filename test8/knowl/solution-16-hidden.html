\begin{equation*} \newcommand{\amp}{\amp }\begin{aligned} \text{Value of first investment}\amp = {10000} e^{{0.037} t} \amp \\ \text{Value of second investment}\amp = {7000} e^{{0.081} t} \amp \\ \text{so}\quad {7000} e^{{0.081} t}\amp = {10000} e^{{0.037} t} \amp \text{set values equal} \\ \frac{e^{{0.081} t}}{e^{{0.037} t}}\amp = \frac{{10000} }{ {7000} } \amp \text{divide} \\ e^{{0.081} t-{0.037} t}\amp = 1.429 \amp \text{simplify} \\ e^{{0.044} t}\amp = 1.429 \amp \\ {0.044} t\amp = \ln 1.429 \amp \text{take logs} \\ t\amp = \frac{\ln 1.429 }{{0.044} } \\ \amp = {8.106} , \end{aligned} \end{equation*}
