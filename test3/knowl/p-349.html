\begin{align*} \int \underbrace{x}_{u}\underbrace{2\cos 2x}_{v'}\,dx\amp = \underbrace{x}_{u}\underbrace{\sin2x}_{v}-\int\underbrace{1}_{u'}\underbrace{\sin2x}_{v}\,dx\\ \text{so}\quad\int2x\cos2x\,dx\amp =uv-\int\sin2x\,dx. \end{align*}
